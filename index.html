function getTaskGroups(sheet, dateStr) { if (!sheet) return []; const defaultSpeedMinutes = (sheet.speed || appData.defaultSpeed) * 60; const speed = appData.dateSpeedOverrides[dateStr] || defaultSpeedMinutes; const groups = []; let currentGroup = []; let currentDuration = 0; const permanentlyCompleted = appData.history.filter(h => h.sheetName === sheet.name && h.permanent === true ).map(h => h.taskIndex); const availableRows = sheet.data .map((row, idx) => ({ row, idx })) .filter(({ idx }) => !permanentlyCompleted.includes(idx)); const totalAvailableDuration = availableRows.reduce((sum, { row }) => sum + (parseFloat(row.duration) || 0), 0); const totalDays = Math.ceil(totalAvailableDuration / speed); if (totalDays <= 0) return []; const targetPerDay = totalAvailableDuration / totalDays; let currentDayDuration = 0; let currentDayRows = []; availableRows.forEach(({ row, idx }) => { const duration = parseFloat(row.duration) || 0; if (currentDayDuration + duration <= targetPerDay) { currentDayRows.push(idx); currentDayDuration += duration; } else { if (currentDayRows.length > 0) { groups.push({ rows: currentDayRows, duration: parseFloat(currentDayDuration.toFixed(2)) }); } currentDayRows = [idx]; currentDayDuration = duration; } }); if (currentDayRows.length > 0) { groups.push({ rows: currentDayRows, duration: parseFloat(currentDayDuration.toFixed(2)) }); } return groups; }
